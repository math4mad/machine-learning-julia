function  plot_heatmap(df)
  matrix=Matrix(df)
  r,c=size(df)
  cors=cor(matrix)
  cats=names(df)

  fig=Figure()
  ax=Axis(fig[1,1],xticklabelrotation = pi/4)
  ax.xticks=(1:c,cats)
  ax.yticks=(1:c,cats)
  heatmap!(ax,cors)

  fig
end
